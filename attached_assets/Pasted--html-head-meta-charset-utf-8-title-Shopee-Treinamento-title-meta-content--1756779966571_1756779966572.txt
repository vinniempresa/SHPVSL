  <script>
   function showPage(page) {
      document.getElementById('home-content').classList.add('hidden');
      document.getElementById('entregas-content').classList.add('hidden');
      document.getElementById('saldo-content').classList.add('hidden');
      document.getElementById(`${page}-content`).classList.remove('hidden');

      // Update active state in navigation
      const navs = ['home', 'entregas', 'saldo'];
      navs.forEach(n => {
        const nav = document.getElementById('nav-' + n);
        if (n === page) {
          nav.querySelector('i').classList.remove('opacity-40');
          nav.querySelector('span').classList.remove('opacity-40');
        } else {
          nav.querySelector('i').classList.add('opacity-40');
          nav.querySelector('span').classList.add('opacity-40');
        }
      });
    }

    // Modal Notificações
    const bellBtn = document.getElementById('bell-btn');
    const modal = document.getElementById('modal-notificacoes');
    const closeModalBtn = document.getElementById('close-modal-btn');
    const modalBg = modal.querySelector('.modal-bg');

    bellBtn.addEventListener('click', function(e) {
      e.stopPropagation();
      modal.classList.remove('hidden');
      document.body.style.overflow = 'hidden';
    });

    function closeModal() {
      modal.classList.add('hidden');
      document.body.style.overflow = '';
    }

    closeModalBtn.addEventListener('click', closeModal);
    modalBg.addEventListener('click', closeModal);

    // Fechar modal com ESC
    document.addEventListener('keydown', function(e) {
      if (!modal.classList.contains('hidden') && e.key === 'Escape') {
        closeModal();
      }
    });
  </script>
